#include <bits/stdc++.h>

using namespace std;
int xs,ys,xd,yd,vis[110][110],mx[4]={-1,1,0,0},my[4]={0,0,-1,1},inf=1e9,n,m;
char a[110][110];
queue<pair<int,int> >qp;
bool inside(int x,int y){
    return x>=1&&x<=n&&y>=1&&y<=m;

}
void bfs( ){
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            vis[i][j]=inf;
    vis[xs][ys]=0;
    qp.push({xs,ys});
    while(!qp.empty()){
        int x=qp.front().first,y=qp.front().second;
        qp.pop();
        for(int i=0;i<4;i++){
            int nx=mx[i]+x;
            int ny=my[i]+y;
            if(inside(nx,ny)&&vis[nx][ny]>=vis[x][y]+1&&a[nx][ny]=='.'){
                vis[nx][ny]=vis[x][y]+1;
                qp.push({nx,ny});
            }
        }
    }

}

int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
        cin>>a[i][j];
                cin>>xs>>ys;
        bfs();
        cout<<vis[n][m];
    return 0;
}
